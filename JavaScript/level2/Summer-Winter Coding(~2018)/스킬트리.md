# 풀이

1. 정규식을 이용한다.
2. 반대의 값을 치환한다.
3. 치환한것이 skill의 첫번째 인덱스와 같는지 확인한다.

# 코드

  function solution(skill, skill_trees) {
    let regex = new RegExp(`[^${skill}]`, 'g');
    
    return arr =
      skill_trees
      .map((x)=>x.replace(regex, ''))
      .filter((x)=>{return skill.indexOf(x)===0})
      .length
  }
