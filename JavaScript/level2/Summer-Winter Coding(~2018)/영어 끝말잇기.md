function solution(n, words) {
let result = [];
let mat = [];
for(let i=0; i<words.length/n; i++){
mat[i]=[]
}
for(let i=0; i<words.length/n; i++){
for(let j=(i*n); j<((i+1)*n); j++){
mat[i].push(words[j])
}
}

    if(words[3].split('').pop() !== words[4].split('')[0]){
        return mat.includes(words[4])
    }

}
